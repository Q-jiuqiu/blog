# 地图上画点

  1. 创建一个简单的地图
<template>
  <div class="map">
    <button @click="draw('point')">绘制点</button>
    <div id="MapBox"></div>
  </div>
</template>

<script>
import { loadModules } from "esri-loader";
export default {
  data() {
    return {
      map: null,
      view: null,
      options: {
        url: "https://js.arcgis.com/4.15/",
        css: "https://js.arcgis.com/4.15/esri/themes/light/main.css"
      },
      drawPen: null,
      drawGraphicLayer: null
    };
  },
  created() {
    this.createdView();
  },
  methods: {
      // 创建简单的地图
    async createdView() {
      let [Map, MapView, GraphicsLayer] = await loadModules(
        ["esri/Map", "esri/views/MapView", "esri/layers/GraphicsLayer"],
        this.options
      );
      this.map = new Map({
        basemap: "streets-navigation-vector"
      });
      this.view = new MapView({
        container: "MapBox",
        zoom: 5,
        map: this.map,
        center: [90, 58]
      });
        
      this.drawGraphicLayer = new GraphicsLayer({ id: "drawGraphicLayer" });
      this.map.add(this.drawGraphicLayer);
    },
    async draw() {
      let [Draw, Graphic, SpatialReference] = await loadModules(
        [
          "esri/views/2d/draw/Draw",
          "esri/Graphic",
          "esri/geometry/SpatialReference"
        ],
        this.options
      );
      this.mapDraw = new Draw({
        view: this.view
      });
      let action = this.mapDraw.create("point");
      action.on("draw-complete", evt => {
        console.log("draw-complete", evt);
        let coordinator = evt.coordinates;
        let spatialReference = new SpatialReference();
        console.log("spatialReference", spatialReference);
        let point = {
          type: "point",
          x: coordinator[0],
          y: coordinator[1],
          spatialReference
        };
        console.log("point", point);
        let graphic = new Graphic({
          geometry: point,
          symbol: {
            type: "simple-marker",
            style: "circle",
            color: "red",
            size: "4px",
            outline: {
              color: [255, 255, 0, 0],
              width: 1
            }
          }
        });
        this.drawGraphicLayer.add(graphic);
      });
    }
  }
};
</script>
