# 创建一个简单的例子

安装依赖：（构建arcgis）

yarn add esri-loader

  1. 导入gis包

    import { loadModules } from "esri-loader";

  2. 创建视图容器

    <div id="viewDiv"></div>

    注意：要设置视图容器的大小

  3. 用JavaScript创建一个简单的2D/3D地图

    //创建2D地图
    let [Map, MapView] = await loadModules([
        "esri/Map", // 加载特定于创建地图的代码
        "esri/views/MapView" // 加载允许以2D模式查看地图的代码
    ]);
    const map = new Map({ // 创建地图
        basemap: "topo-vector" // 底图,多个值可选
    });
    let view = new MapView({ // 加载2D地图
        container: "viewDiv", // 用于显示地图id(地图容器的id)
        map: map, // 地图
        zoom: 4, // 地图缩放级别
        center: [15, 65]  // 地图视图中心
    });
//创建3D地图
    let [Map, SceneView] = await loadModules([
        "esri/Map", // 加载特定于创建地图的代码
        "esri/views/SceneView" // 加载允许以3D模式查看地图的代码
    ]);
    const map = new Map({ // 创建地图
        basemap: "topo-vector" // 底图,多个值可选
    });
    let view = new MapView({ // 加载2D地图
        container: "viewDiv", // 用于显示地图id(地图容器的id)
        map: map, // 地图
        zoom: 4, // 地图缩放级别
        center: [15, 65]  // 地图视图中心
    });
