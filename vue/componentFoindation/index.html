# 组件基础

  1. 全局组件

    // ---在main.js中---
// 注册全局自定义组件
Vue.component("allButton", {
  data() {
    return { count: 0 };
  },
  template: '<button @click="count++">{{count}}</button>'
});

new Vue({
  el: "#app"
});

  2. 局部组件

    --- 第一种写法 ---
<script>
export default {
  components: {
    myButton: {
      template: '<button @click="add">{{count}}</button>',
      data() {
        return {
          count: 1
        };
      },
      methods: {
        add() {
          this.count++;
        }
      }
    }
  }
};
</script>

--- 第二种写法 ---
<script>
import myA from "./components/A";
export default {
  data() {
    return { param1: "jhh", param2: "123" };
  },
  components: {
    myA
  }
};
</script>

因为你每用一次组件，就会有一个它的新实例被创建，所以在复用组件的时候他们自己的值(data)是相互独立的。

# 为什么data是一个函数?

组件是一个具有单独自身功能模块的封装，这个模块有属于自己的HTML模板，也应该有属于自己的数据data，组件也有自己的方法methods。data属性必须是一个函数，并且这个函数返回一个对象，在对象内部保存数据

// 模拟data不是函数
let obj = {
  name: "name"
};

function objFun() {
  return obj;
}
let obj1 = objFun();
let obj2 = objFun();
// 由于对象存在堆中的,而变量仅仅是保存了对象的引用地址（浅拷贝）,
// 所以obj1和obj2保存的都是obj对象的引用,
// 即obj、obj1、obj2任意一个改变都会影响其他的。
// 而这样的效果违背了组件复用而组件内的值独立的初衷
// 初衷--每个实例可以维护一份被返回对象的独立的拷贝

obj1.name = "zhang";
console.log(obj1.name, obj2.name); //输出结果为:zhang zhang


// 模拟data是函数
function person() {
  return {
    name: "name"
  };
}

let person1 = person();
let person2 = person();

person1.name = "zhang";
console.log(person1.name, person2.name);//输出结果为:zhang name

# 局部组件注册

全局注册往往是不够理想的。比如，如果你使用一个像 webpack 这样的构建系统，全局注册所有的组件意味着即便你已经不再使用一个组件了，它仍然会被包含在你最终的构建结果中。这造成了用户下载的 JavaScript 的无谓的增加。

import myA from "./components/A";
import myB from "./components/B";
export default {
  name: "Store",
  components: {
    myA,
    myB
  }
};

# 自动全局注册组件

方法：

require.context(参数1, 参数2, 参数3）

参数1：需要注册的组件所在的文件路径——String

参数1：需要挂载的组件所在的文件路径 String

参数2：是否读取文件夹下的子文件夹——boolean

参数3：匹配规则，只有满足匹配规则的文件的文件路径才会被获取到——正则表达式

keys()：遍历返回键名，返回值是一个数组

import Vue from "vue";

const files = require.context("../components", true, /\.vue$/);
// files是一个方法

files.keys().forEach(item => {
  const name = item.replace(/^\.\/(.*\/)?/, "").replace(/\.vue$/, "");
    // files(item) 获取组件配置
    // 全局注册
  Vue.component(name, files(item).default || files(item));
});

# 自动引入局部组件

结合动态组件使用(component + :is)

const files = require.context("../components", true, /\.vue$/);

/**
 * @description: 根据组件名自动引入对应组件
 * @param {name} 组件名
 */

const getCompByName = name => {
  let comp = null;
  files.keys().some(item => {

    const fileName = item.replace(/^\.\/(.*\/)?/, "").replace(/\.vue/, "");

    if (name === fileName) {
      comp = files(item);
      return true;
    }
  });
  return comp.default || comp;
};
export default getCompByName;


<template>
  <div>
    <component :is="Comp"></component>
  </div>
</template>

<script>
import getCompByName from "./utils/autoRegister";

export default {
  computed: {
    Comp() {
      // return getCompByName("Header");
      return getCompByName("Body");
    }
  }
};
</script>

# 单向数据流

单向数据流：父组件向子组件传值，而子组件无法修改父组件传来的值，也无法将之返回给父组件。每次父级组件发生变更时，子组件中所有的 prop 都将会刷新为最新的值。

父组件：

<template>
  <div>
    <son :compA="a"></son>
  </div>
</template>

<script>
import son from "./Prop/Prop";

export default {
  components: {
    son
  },
  data() {
    return {
      a: 111
    };
  },
};
</script>

子组件：

<template>
  <div class="Prop">
    This is Prop.
    <div>
      {{ this.compA }}
    </div>
  </div>
</template>

<script>
export default {
    //写法一:
  props: ["compA"], // 用于接收父组件传过来的值
    //写法二:带有验证功能
  props: {
    compA: {
      type: Number, // 规定父组件传过来的值得类型
      default: 0 // 默认值
    }
  }
};
</script>

# Prop验证

Vue.component('my-component', {
  props: {
    // 基础的类型检查 (`null` 和 `undefined` 会通过任何类型验证)
    propA: Number,
    // 多个可能的类型
    propB: [String, Number],
    // 必填的字符串
    propC: {
      type: String,
      required: true
    },
    // 带有默认值的数字
    propD: {
      type: Number,
      default: 100
    },
    // 带有默认值的对象
    propE: {
      type: Object,
      // 对象或数组默认值必须从一个工厂函数获取
      default: function () {
        return { message: 'hello' }
      }
    },
    // 自定义验证函数
    propF: {
      validator: function (value) {
        // 这个值必须匹配下列字符串中的一个
        return ['success', 'warning', 'danger'].indexOf(value) !== -1
      }
    }
  }
})

注意：

那些 prop 会在一个组件实例创建之前进行验证，所以实例的 property (如 data、computed 等) 在 default 或 validator 函数中是不可用的。
