# grid布局 (opens new window)

# 基本概念

# 容器和项目

采用网格布局的区域，称为"容器"（container）。容器内部采用网格定位的子元素，称为"项目"（item）。

注意：项目只能是容器的顶层子元素，不包含项目的子元素

# 行和列

# 实例

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>grid布局-1</title>
    <style>
      .grid {
        width: 300px;
        height: 300px;
        background-color: pink;
        margin-bottom: 1px;
        /* display: inline-grid;行内grid */
          
        /* 写法一: */
        /* grid-template-columns: 100px 100px 100px;
              grid-template-rows: 100px; */
          
        /* 写法二： */
        /* grid-template-columns: 33.33% 33.33% 33.33%;
              grid-template-rows: 33.33%; */
          
        /* 写法三 */
        /* grid-template-columns: repeat(3, 33.33%); 
          列间隔33.33%重复3次 */
        /* grid-template-columns: repeat(2, 100px 50px); */
        /* grid-template-columns: repeat(auto-fill, 110px);
          每列宽度100px，然后自动填充，直到容器不能放置更多的列。 */
        /* grid-template-columns: 1fr 2fr; 
          相当于容器宽度等分成三分,一份占1一份占2 */
        /* grid-template-columns: 150px 1fr 2fr; 
          相当于第一列占150px,第二列占剩下(300-150)px的1/3,第三列占2/3 */
        /* grid-template-columns: 1fr 2fr minmax(100px, 1fr); 
          第三列的宽度的取值范围 */
        /* grid-template-columns: 1fr 2fr auto;
          auto关键字表示由浏览器自己决定长度。
          在没有min-width的情况下取最大值 */
        /* grid-template-rows: 33.33%; */

        /* 写法三 */ /*网格线名称*/
        display: grid;
        grid-template-columns: [c1] 100px [c2] 100px [c3] auto [c4];
        grid-template-rows: [r1] 100px [r2] 100px [r3] auto [r4];
        grid-template-columns: repeat(3, 1fr);
        grid-template-rows: repeat(3, 1fr);

        /* grid-row-gap: 10px; 行间距 */
        /* grid-column-gap: 10px; 列间距 */
        /* grid-gap: 10px 20px; 行、列间距 */
        grid-gap: 10px; /*行、列间距*/

        grid-template-areas: "header header ." "aside main main" "footer footer footer";
        /* 把grid划分区域,每个区域都有自己的名字,"."表示该区域不需要利用 */

        /* grid-auto-flow: column; */
        /* grid-auto-flow: row dense; 先行后列 */
        /* grid-auto-flow: column dense; 先列后行 */
        /* justify-items: baseline; 行对齐方式 */
        /* align-items: center;列对齐方式 */
        /* place-items: <align-items> <justify-items>; 
          简写行、列的对齐方式*/
        /* place-items: start end; 省略第二个值就会默认等于第一个*/
        /* grid-auto-rows: 50px; 超出的网格线的行的高度 */
        /* grid-auto-columns: 50px; 超出的网格线的lie的高度 */
      }

      .item {
        /* width: 100px;
              height: 100px; */
        background-color: plum;
        /* display: inline-block; */
        /* 父元素是采用grid布局所以项目的display都采用默认的,设置不起效果 */
      }
      .item:nth-child(odd) {
        background-color: palegoldenrod;
      }
      .item:nth-child(even) {
        background-color: yellowgreen;
      }
      .item.header {
        /* 利用  grid-area:区域名使用该区域 */
        grid-area: header;
        background-color: #ea4335;
      }
      .item.aside {
        grid-area: aside;
        background-color: #ff9800;
      }
      .item.main {
        grid-area: main;
        background-color: #fbbc05;
      }
      .item.footer {
        grid-area: footer;
        background-color: #34a853;
      }
      .test {
        /* 指定项目的行、列的起始、和结束的分别对齐哪个网格线,
          可以使用指定网格线名称 */
        /* grid-column-start: 2;
        grid-column-end: span 2; 跨越两行/列
        grid-row-start: 1;
        grid-row-end: 3; */
        /* grid-column-start: c1;
        grid-column-end: span 2;
        grid-row-start: r1;
        grid-row-end: r3; */
        background: #b03532;
        /* 起始线和终止线的简写 */
        grid-column: 1 / span 2;
        grid-row: 1 / span 2;

        /* 项目内的对齐 */
        justify-self: center; /*行*/
        align-self: center; /*lie*/
        place-self: center center;
      }
    </style>
  </head>
  <body>
    <span>foo</span>
    <div class="grid">
      <div class="item header">1</div>
      <div class="item aside">2</div>
      <div class="item main">3</div>
      <div class="item footer">4</div>
      <div class="item">5</div>
      <div class="item">6</div>
      <div class="item">7</div>
      <div class="item">8</div>
      <div class="item test">9</div>
    </div>
    <span>bar</span>
  </body>
</html>
